Домашнее задание №8

Написать класс View. Все наследуемые от него классы должны иметь возможность легко развешивать события на свои дочерние элементы и на себя.

Пример:

function MyView () {
  // ...
}

inherits(View, MyView);

MyView.prototype.events = {
  “click”: “doSmth”,
  “click .a”: “doA”,
  “click .b”: “doB”
};

MyView.prototype.doSmth = function(){
  console.log(“doSmth”);
};

MyView.prototype.doA = function(){
  console.log(“doA”);
};

MyView.prototype.doB = function(){
  console.log(“doB”);
};

Элементы:

el     a     b  

Инициализация:

var view = new MyView({
  el: document.boby.getElementsByClassName(‘el’)[0]
});
